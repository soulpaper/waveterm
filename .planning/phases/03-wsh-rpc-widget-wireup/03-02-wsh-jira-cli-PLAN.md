         ```go
         func jiraRefreshRun(cmd *cobra.Command, args []string) (rtnErr error) {
             defer func() {
                 sendActivity("jira-refresh", rtnErr == nil)
             }()

             opts := &wshrpc.RpcOpts{Timeout: jiraRefreshTimeout * 1000}
             if tabId := os.Getenv("WAVETERM_TABID"); tabId != "" {
                 opts.Route = wshutil.MakeTabRouteId(tabId)
             }
             // empty Route → default wavesrv route (refresh works outside a tab)

             rtn, err := wshclient.JiraRefreshCommand(RpcClient, wshrpc.CommandJiraRefreshData{}, opts)
             if err != nil {
                 fmt.Fprintln(os.Stderr, err.Error())
                 os.Exit(exitCodeForError(err))
             }

             if jiraRefreshJSON {
                 out, marshalErr := json.MarshalIndent(rtn, "", "  ")
                 if marshalErr != nil {
                     return fmt.Errorf("marshal rtn: %w", marshalErr)
                 }
                 fmt.Println(string(out))
                 return nil
             }

             fmt.Println(formatRefreshSummary(rtn))
             return nil
         }
         ```
       - Implement pure helpers:

         func exitCodeForError(err error) int {
             if err == nil {
                 return 0
             }
             msg := err.Error()
             switch {
             case strings.HasPrefix(msg, "인증 실패"):
                 return 1
             case strings.HasPrefix(msg, "설정 파일이 없습니다"):
                 return 2
             default:
                 return 3
             }
         }

         func formatRefreshSummary(rtn wshrpc.CommandJiraRefreshRtnData) string {
             elapsed := time.Duration(rtn.ElapsedMs) * time.Millisecond
             return fmt.Sprintf("Fetched %d issues (%d attachments, %d comments) in %.1fs → %s",

                 elapsed.Seconds(), rtn.CachePath)
         }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user shell → wsh binary | User types command; argv is parsed by cobra. Only `--timeout` takes a numeric value (bounded by Go int parsing). |
| wsh → wavesrv (RPC) | CLI invokes the same handler widget does. No additional validation needed — handler is the source of truth. |
| RPC error → stderr | Error.Error() is printed verbatim to stderr. These strings come from Plan 01's mapJiraError which is already sanitized (T-03-01). |
